    {
        let sum = [0, 0];
        for (let i = 0; i < N; i++){
            if(Arr[i] % 2 === 0){
            sum[0] += Arr[i];

        }

Que 9: Unique Color Shirt

Ans:
        
function Unique_Shirts (arr,N) {

    let uniqueElement = 0;
    for(let j = 0; j < N; j++){
      let count = 0;
      let key = arr[j];
      for(let i = 0; i < N; i++){
        if(key == arr[i]){
          count++;
        }
        
      }
      if(count == 1){
          uniqueElement++;
        }
    }
    return uniqueElement;
}

Que 10: Min and Max

Ans:
        
function arrayMin(arr) {
    let min = arr[0];
   for(let i = 0; i < arr.length; i++){
     if(arr[i] < min){
       min = arr[i];
     }
   }
   return min;
 }
 
 function arrayMax(arr) {
   let max = arr[0];
   for(let i = 0; i < arr.length; i++){
     if(arr[i] > max){
       max = arr[i];
     }
   }
   return max;
 }

Ans:
        
function Birthday_Game(arr,D ,M) {
 
    let count=0;
    for(i=0;i <= arr.length-M;i++){
      let sum=0;
      for(j=0;j < M;j++){
        sum = sum + arr[i+j];
      }
      if(sum == D){
        count++;
      }
    }
    return count;
    }
